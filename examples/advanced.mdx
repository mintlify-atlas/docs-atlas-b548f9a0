## Full-Featured Auto-Scrolling Carousel

Combine auto-scroll, custom easing, and styled icons for a production-ready carousel:

```jsx
import { Carousel } from 'react-any-carousal';

function AdvancedCarousel() {
  const slides = [
    {
      id: 1,
      image: '/slides/hero-1.jpg',
      title: 'Premium Quality',
      description: 'Crafted with attention to every detail',
      cta: 'Shop Now',
    },
    {
      id: 2,
      image: '/slides/hero-2.jpg',
      title: 'Fast Delivery',
      description: 'Get your order within 48 hours',
      cta: 'Learn More',
    },
    {
      id: 3,
      image: '/slides/hero-3.jpg',
      title: 'Eco Friendly',
      description: 'Sustainable materials, better future',
      cta: 'Discover',
    },
  ];

  const customIcon = (
    <svg width="20" height="20" viewBox="0 0 20 20" fill="none" stroke="currentColor">
      <path d="M7 4l6 6-6 6" strokeWidth="2.5" strokeLinecap="round" strokeLinejoin="round" />
    </svg>
  );

  return (
    <Carousel 
      autoSlideInterval={5000}
      duration={800}
      scrollEasing="cubic-bezier(0.34, 1.56, 0.64, 1)" // Bounce effect
      scrollSnapType="start"
      scrollOffset={1200}
      theme="dark"
      iconOptions={{
        icon: customIcon,
        iconStyles: {
          color: 'white',
          backgroundColor: 'rgba(0, 0, 0, 0.5)',
          backdropFilter: 'blur(10px)',
          padding: '1rem',
          borderRadius: '50%',
          border: '2px solid rgba(255, 255, 255, 0.2)',
          boxShadow: '0 4px 16px rgba(0, 0, 0, 0.3)',
          transition: 'all 0.3s ease',
        },
      }}
    >
      {slides.map((slide) => (
        <div key={slide.id} className="hero-slide-advanced">
          <img src={slide.image} alt={slide.title} />
          <div className="overlay" />
          <div className="content">
            <h1>{slide.title}</h1>
            <p>{slide.description}</p>
            <button className="cta">{slide.cta}</button>
          </div>
        </div>
      ))}
    </Carousel>
  );
}
```

## Dynamic Product Showcase

Build an e-commerce carousel with filtering and dynamic content:

```jsx
import { Carousel } from 'react-any-carousal';
import { useState } from 'react';

function ProductShowcase() {
  const [category, setCategory] = useState('all');
  
  const products = [
    { id: 1, name: 'Wireless Headphones', price: 199, category: 'audio', image: '/products/headphones.jpg' },
    { id: 2, name: 'Smart Watch', price: 299, category: 'wearables', image: '/products/watch.jpg' },
    { id: 3, name: 'Laptop Stand', price: 79, category: 'accessories', image: '/products/stand.jpg' },
    { id: 4, name: 'Bluetooth Speaker', price: 149, category: 'audio', image: '/products/speaker.jpg' },
    { id: 5, name: 'Phone Case', price: 29, category: 'accessories', image: '/products/case.jpg' },
  ];

  const filteredProducts = category === 'all' 
    ? products 
    : products.filter(p => p.category === category);

  const chevronIcon = (
    <svg width="16" height="16" viewBox="0 0 16 16" fill="currentColor">
      <path d="M6 4l4 4-4 4" stroke="currentColor" strokeWidth="2" fill="none" strokeLinecap="round" />
    </svg>
  );

  return (
    <div className="product-showcase">
      <div className="filter-bar">
        <button 
          className={category === 'all' ? 'active' : ''}
          onClick={() => setCategory('all')}
        >
          All Products
        </button>
        <button 
          className={category === 'audio' ? 'active' : ''}
          onClick={() => setCategory('audio')}
        >
          Audio
        </button>
        <button 
          className={category === 'wearables' ? 'active' : ''}
          onClick={() => setCategory('wearables')}
        >
          Wearables
        </button>
        <button 
          className={category === 'accessories' ? 'active' : ''}
          onClick={() => setCategory('accessories')}
        >
          Accessories
        </button>
      </div>

      <Carousel 
        key={category} // Reset carousel when category changes
        scrollSnapType="start"
        scrollOffset={290}
        duration={500}
        scrollEasing="cubic-bezier(0.25, 1, 0.5, 1)"
        iconOptions={{
          icon: chevronIcon,
          iconStyles: {
            color: '#1f2937',
            backgroundColor: 'white',
            padding: '0.875rem',
            borderRadius: '50%',
            boxShadow: '0 4px 12px rgba(0, 0, 0, 0.1)',
          },
        }}
      >
        {filteredProducts.map((product) => (
          <div key={product.id} className="product-card-advanced">
            <div className="product-image">
              <img src={product.image} alt={product.name} />
              <span className="category-badge">{product.category}</span>
            </div>
            <div className="product-info">
              <h3>{product.name}</h3>
              <p className="price">${product.price}</p>
              <button className="add-to-cart">Add to Cart</button>
            </div>
          </div>
        ))}
      </Carousel>
    </div>
  );
}
```

## Lazy-Loaded Image Carousel

Optimize performance with lazy loading for large image sets:

```jsx
import { Carousel } from 'react-any-carousal';
import { useState, useRef, useEffect } from 'react';

function LazyImage({ src, alt }) {
  const [isLoaded, setIsLoaded] = useState(false);
  const [isInView, setIsInView] = useState(false);
  const imgRef = useRef(null);

  useEffect(() => {
    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          setIsInView(true);
          observer.disconnect();
        }
      },
      { rootMargin: '200px' }
    );

    if (imgRef.current) {
      observer.observe(imgRef.current);
    }

    return () => observer.disconnect();
  }, []);

  return (
    <div ref={imgRef} className="lazy-image-container">
      {isInView && (
        <img
          src={src}
          alt={alt}
          onLoad={() => setIsLoaded(true)}
          className={isLoaded ? 'loaded' : 'loading'}
        />
      )}
      {!isLoaded && <div className="image-placeholder" />}
    </div>
  );
}

function LazyLoadCarousel() {
  const images = Array.from({ length: 20 }, (_, i) => ({
    id: i + 1,
    src: `/gallery/image-${i + 1}.jpg`,
    alt: `Gallery image ${i + 1}`,
  }));

  return (
    <Carousel 
      scrollSnapType="center"
      scrollOffset={420}
      duration={600}
      scrollEasing="cubic-bezier(0.4, 0, 0.2, 1)"
      theme="dark"
    >
      {images.map((image) => (
        <div key={image.id} className="gallery-item">
          <LazyImage src={image.src} alt={image.alt} />
        </div>
      ))}
    </Carousel>
  );
}
```

## Custom Easing Examples

Experiment with different easing functions for unique scroll behaviors:

```jsx
import { Carousel } from 'react-any-carousal';
import { useState } from 'react';

function EasingShowcase() {
  const [easing, setEasing] = useState('cubic-bezier(0.25, 1, 0.5, 1)');
  
  const easingOptions = [
    { name: 'Smooth', value: 'cubic-bezier(0.25, 1, 0.5, 1)' },
    { name: 'Bounce', value: 'cubic-bezier(0.34, 1.56, 0.64, 1)' },
    { name: 'Fast Start', value: 'cubic-bezier(0.7, 0, 0.3, 1)' },
    { name: 'Slow Start', value: 'cubic-bezier(0.1, 0, 0.9, 1)' },
    { name: 'Elastic', value: 'cubic-bezier(0.68, -0.55, 0.265, 1.55)' },
  ];

  return (
    <div className="easing-showcase">
      <h2>Try Different Easing Functions</h2>
      <div className="easing-controls">
        {easingOptions.map((option) => (
          <button
            key={option.name}
            className={easing === option.value ? 'active' : ''}
            onClick={() => setEasing(option.value)}
          >
            {option.name}
          </button>
        ))}
      </div>

      <Carousel 
        key={easing} // Force re-render when easing changes
        scrollSnapType="center"
        scrollOffset={350}
        duration={1000}
        scrollEasing={easing}
      >
        {[1, 2, 3, 4, 5].map((num) => (
          <div key={num} className="easing-slide">
            <h3>Slide {num}</h3>
            <p>Current easing: {easingOptions.find(e => e.value === easing)?.name}</p>
          </div>
        ))}
      </Carousel>
    </div>
  );
}
```

## Performance Optimization Tips

<AccordionGroup>
  <Accordion title="Large Datasets">
    When working with many items:
    - Implement lazy loading for images and heavy content
    - Use virtualization for extremely large lists (100+ items)
    - Consider pagination instead of loading everything
    - Debounce rapid scroll events if adding custom handlers
  </Accordion>

  <Accordion title="Animation Performance">
    Optimize smooth scrolling:
    - Keep `duration` between 400-800ms for best performance
    - Avoid heavy computations during scroll animations
    - Use CSS transforms for item animations (GPU accelerated)
    - Test on lower-end devices to ensure smooth performance
  </Accordion>

  <Accordion title="Memory Management">
    Prevent memory leaks:
    - Clean up event listeners in useEffect
    - Avoid storing large objects in carousel item state
    - Use React.memo for expensive child components
    - Unmount carousels when not visible (tabs, modals)
  </Accordion>

  <Accordion title="Image Optimization">
    Best practices for images:
    - Serve responsive images with srcset
    - Use modern formats (WebP, AVIF) with fallbacks
    - Implement progressive loading (blur-up technique)
    - Compress images appropriately (aim for under 200KB each)
  </Accordion>
</AccordionGroup>

## Error Handling

Handle edge cases gracefully:

```jsx
import { Carousel } from 'react-any-carousal';
import { useState, useEffect } from 'react';

function RobustCarousel() {
  const [items, setItems] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    fetchCarouselItems()
      .then(data => {
        setItems(data);
        setLoading(false);
      })
      .catch(err => {
        setError(err.message);
        setLoading(false);
      });
  }, []);

  if (loading) {
    return <div className="carousel-loading">Loading amazing content...</div>;
  }

  if (error) {
    return <div className="carousel-error">Failed to load content: {error}</div>;
  }

  if (items.length === 0) {
    return <div className="carousel-empty">No items to display</div>;
  }

  return (
    <Carousel 
      scrollSnapType="center"
      scrollOffset={420}
      duration={600}
    >
      {items.map((item) => (
        <div key={item.id} className="carousel-item">
          {item.content}
        </div>
      ))}
    </Carousel>
  );
}

// Mock fetch function
function fetchCarouselItems() {
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve([
        { id: 1, content: 'Item 1' },
        { id: 2, content: 'Item 2' },
        { id: 3, content: 'Item 3' },
      ]);
    }, 1000);
  });
}
```

## Common Easing Functions Reference

```jsx
// Smooth and natural
scrollEasing="cubic-bezier(0.25, 1, 0.5, 1)"

// Bounce effect
scrollEasing="cubic-bezier(0.34, 1.56, 0.64, 1)"

// Fast start, slow end
scrollEasing="cubic-bezier(0.7, 0, 0.3, 1)"

// Slow start, fast end
scrollEasing="cubic-bezier(0.1, 0, 0.9, 1)"

// Elastic overshoot
scrollEasing="cubic-bezier(0.68, -0.55, 0.265, 1.55)"

// Linear (constant speed)
scrollEasing="cubic-bezier(0, 0, 1, 1)"

// Ease in-out (default browser)
scrollEasing="cubic-bezier(0.42, 0, 0.58, 1)"
```

<Tip>
  Visit [cubic-bezier.com](https://cubic-bezier.com) to visualize and create custom easing functions.
</Tip>
